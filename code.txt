function sendTelegramOTP(chatId, otp) {
  try {
    if (!chatId || chatId === "" || chatId === "undefined") {
      return { status: false, errorDetail: "Chat ID kosong." };
    }
    
    let message = `🔑 *Kode OTP Login Sym_Pro*\n\nKode verifikasi Anda adalah: *${otp}*\nBerlaku selama 5 menit. Jangan berikan kode ini kepada siapa pun.`;
    let url = `https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`;
    
    let payload = {
      method: 'post',
      contentType: 'application/json',
      payload: JSON.stringify({
        chat_id: chatId,
        text: message,
        parse_mode: 'Markdown'
      }),
      muteHttpExceptions: true
    };
    
    let response = UrlFetchApp.fetch(url, payload);
    let result = JSON.parse(response.getContentText());
    
    if (result.ok) {
      return { status: true };
    } else {
      return { status: false, errorDetail: result.description || "Gagal dari Telegram API" };
    }
  } catch (e) {
    return { status: false, errorDetail: e.message };
  }
}

function processLogin(username, password) {
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName("Users");
    if (!sheet) {
      return { success: false, message: "Sheet 'Users' tidak ditemukan pada Spreadsheet!" };
    }
    
    const dataRange = sheet.getDataRange().getValues();
    const inputUser = String(username).trim();
    const inputPass = String(password).trim();
    
    for (let i = 1; i < dataRange.length; i++) {
      const dbEmail = String(dataRange[i][2] || "").trim();
      const dbUsername = String(dataRange[i][3] || "").trim();
      const dbPass = String(dataRange[i][4] || "").trim();
      
      const dbRole = dataRange[i][6] ? String(dataRange[i][6]).trim().toLowerCase() : 'user';
      const dbStatus = dataRange[i][7] ? String(dataRange[i][7]).trim().toLowerCase() : '';
      
      if ((dbEmail === inputUser || dbUsername === inputUser) && dbPass === inputPass) {
        
        if (dbStatus !== 'active') {
          return { success: false, message: "Akun Anda belum di-approve oleh Superadmin." };
        }
        
        const chatId = String(dataRange[i][5] || "").trim(); 
        if (!chatId || chatId === "" || chatId === "undefined") {
          return { success: false, message: "Chat ID Telegram tidak valid." };
        }
        
        const otp = Math.floor(100000 + Math.random() * 900000).toString();
        
        const cache = CacheService.getScriptCache();
        const sessionData = { otp: otp, role: dbRole, name: String(dataRange[i][1] || inputUser) };
        cache.put(inputUser, JSON.stringify(sessionData), 300);
        
        const sendStatus = sendTelegramOTP(chatId, otp);
        
        if(sendStatus.status === true) {
          return { success: true, requireOtp: true, message: "OTP telah dikirim ke Telegram Anda." };
        } else {
          return { success: false, message: "Gagal Kirim OTP! " + (sendStatus.errorDetail || "Pastikan sudah tekan Start di Bot Telegram.") };
        }
      }
    }
    return { success: false, message: "Username/Email atau Password salah!" };
  } catch(e) {
    return { success: false, message: "Error Server: " + e.message };
  }
}

function verifyOTP(username, inputOtp) {
  try {
    const cache = CacheService.getScriptCache();
    const cachedString = cache.get(username);
    
    if (cachedString) {
      const cachedData = JSON.parse(cachedString);
      
      if (cachedData.otp === inputOtp) {
        cache.remove(username);
        return { success: true, role: cachedData.role, name: cachedData.name, message: "Login Berhasil!" };
      }
    }
    
    return { success: false, message: "OTP salah atau sudah kedaluwarsa!" };
  } catch(e) {
    return { success: false, message: "Error: " + e.message };
  }
}

function getSheetDataByName(sheetName) {
  try {
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(sheetName);
    if (!sheet) return { success: false, message: "Sheet '" + sheetName + "' tidak ditemukan." };
    
    const data = sheet.getDataRange().getValues();
    if (!data || data.length <= 1) {
      const headers = (data && data.length > 0) ? data[0] : [];
      return { success: true, headers: headers, rows: [] };
    }
    
    const headers = data[0];
    const rows = [];
    
    for (let i = 1; i < data.length; i++) {
      let row = data[i];
      let hasContent = row.some(cell => cell !== undefined && cell !== null && String(cell).trim() !== "");
      
      if (hasContent) {
        let rowStr = row.join(" ").toUpperCase();
        let isHeaderFormulaRow = rowStr.includes("WORKORDER") && rowStr.includes("CUSTOMER");
        
        if (!isHeaderFormulaRow) {
          let cleanRow = row.map(cell => {
            if (cell instanceof Date) {
              return cell.toLocaleString('id-ID');
            }
            return cell !== undefined && cell !== null ? String(cell) : "";
          });
          rows.push(cleanRow);
        }
      }
    }

    return { success: true, headers: headers, rows: rows };
  } catch(e) {
    return { success: false, message: e.message };
  }
}

function updateDailyOrderData(payload) {
  try {
    const targetSheetName = payload.sheetName || "List Daily Order";
    const sheet = SpreadsheetApp.openById(SHEET_ID).getSheetByName(targetSheetName);
    if (!sheet) return { success: false, message: "Sheet '" + targetSheetName + "' tidak ditemukan." };

    const data = sheet.getDataRange().getValues();
    const headers = data[0].map(h => String(h).trim().toLowerCase());
    
    let targetRowIdx = -1;
    let woCol = headers.findIndex(h => h.includes('workorder'));
    let trCol = headers.findIndex(h => h === 'order id tsel');
    if (trCol === -1) {
      trCol = headers.findIndex(h => h.includes('sc order no') || h.includes('csrm'));
    }

    for (let i = 1; i < data.length; i++) {
      let wVal = woCol !== -1 ? String(data[i][woCol]).trim() : "";
      let tVal = trCol !== -1 ? String(data[i][trCol]).trim() : "";
      if ((payload.oldWorkorder && wVal === payload.oldWorkorder) || (payload.oldTrackId && tVal === payload.oldTrackId) || (payload.workorder && wVal === payload.workorder)) {
        targetRowIdx = i + 1;
        break;
      }
    }

    if (targetRowIdx === -1) return { success: false, message: "Order tidak ditemukan di Sheet " + targetSheetName };

    const updateMap = {
      "crm order type": payload.crmOrderType,
      "workorder": payload.workorder,
      "order id tsel": payload.trackId,
      "sc order no/track id/csrm no": payload.trackId,
      "service no.": payload.serviceNo,
      "package_info": payload.packageInfo,
      "leader": payload.leader,
      "member": payload.member,
      "errorcode": payload.errorCode,
      "suberrorcode": payload.subErrorCode,
      "engineermemo (sa)": payload.engineerMemo
    };

    if (payload.action === 'assign' && payload.assignTime) {
      updateMap["assign date"] = payload.assignTime;
      updateMap["waktu assign"] = payload.assignTime;
      updateMap["assigned date"] = payload.assignTime;
    }

    for (let colName in updateMap) {
      if (updateMap[colName] !== undefined && updateMap[colName] !== null) {
        let idx = headers.findIndex(h => h === colName || h === colName.toLowerCase());
        if (idx !== -1) {
          sheet.getRange(targetRowIdx, idx + 1).setValue(updateMap[colName]);
        }
      }
    }

    if (payload.action === 'assign') {
      sendTelegramNotification(payload.trackId || payload.workorder, payload.leader, payload.member, payload.timName);
    }

    // Success notification text updated per requirement
    return { success: true, message: "Data Berhasil disimpan!" };
  } catch(e) {
    return { success: false, message: "Gagal update: " + e.message };
  }
}

function syncNewOrdersFromUpdateData() {
  try {
    var ss = SpreadsheetApp.openById(SHEET_ID);
    
    var sourceSheet = ss.getSheetByName("Update Data");
    var destSheet = ss.getSheetByName("List Daily Order");
    var sheetKendala = ss.getSheetByName("Order Kendala");
    var sheetCompleted = ss.getSheetByName("Order Completed");
    
    if (!sourceSheet || !destSheet) {
      return { success: false, message: "Sheet 'Update Data' atau 'List Daily Order' tidak ditemukan!" };
    }
    
    var KEY_NAMES = ["SC Order No/Track ID/CSRM No", "Order ID TSEL"];

    var destLastCol = destSheet.getLastColumn() || 10;
    var destHeaders = destSheet.getRange(1, 1, 1, destLastCol).getValues()[0];
    
    var destKeyIndex = -1;
    for (var h = 0; h < destHeaders.length; h++) {
      var hStr = String(destHeaders[h]).trim().toLowerCase();
      if (hStr.includes("sc order no") || hStr.includes("csrm") || hStr.includes("order id tsel")) {
        destKeyIndex = h + 1;
        break;
      }
    }
    if (destKeyIndex === -1) destKeyIndex = 4;

    var destLastRow = 0;
    var maxDestRows = destSheet.getLastRow();
    if (maxDestRows >= 1) {
      var checkRange = destSheet.getRange(1, destKeyIndex, maxDestRows, 1).getValues();
      for (var i = checkRange.length - 1; i >= 0; i--) {
        var valKeyCheck = String(checkRange[i][0]).trim();
        if (valKeyCheck !== "" && !KEY_NAMES.some(k => k.toLowerCase() === valKeyCheck.toLowerCase())) {
          destLastRow = i + 1;
          break;
        }
      }
    }
    if (destLastRow === 0) destLastRow = 1;

    var existingKeys = {};
    
    // 1. Record keys from 'List Daily Order'
    if (destLastRow >= 2) {
      var destKeys = destSheet.getRange(2, destKeyIndex, destLastRow - 1, 1).getValues();
      for (var k = 0; k < destKeys.length; k++) {
        var kVal = String(destKeys[k][0]).trim();
        if (kVal && !KEY_NAMES.some(kn => kn.toLowerCase() === kVal.toLowerCase())) {
          existingKeys[kVal] = true;
        }
      }
    }
    
    // 2. Record keys from 'Order Kendala'
    if (sheetKendala && sheetKendala.getLastRow() >= 2) {
      var kHeaders = sheetKendala.getRange(1, 1, 1, sheetKendala.getLastColumn()).getValues()[0];
      var kKeyIdx = -1;
      for (var kh = 0; kh < kHeaders.length; kh++) {
        var khStr = String(kHeaders[kh]).trim().toLowerCase();
        if (khStr.includes("sc order no") || khStr.includes("csrm") || khStr.includes("order id tsel")) {
          kKeyIdx = kh + 1;
          break;
        }
      }
      if (kKeyIdx === -1) kKeyIdx = 4;
      var kendalaKeys = sheetKendala.getRange(2, kKeyIdx, sheetKendala.getLastRow() - 1, 1).getValues();
      for (var ki = 0; ki < kendalaKeys.length; ki++) {
        var kValK = String(kendalaKeys[ki][0]).trim();
        if (kValK && !KEY_NAMES.some(kn => kn.toLowerCase() === kValK.toLowerCase())) {
          existingKeys[kValK] = true;
        }
      }
    }
    
    // 3. Record keys from 'Order Completed'
    if (sheetCompleted && sheetCompleted.getLastRow() >= 2) {
      var cHeaders = sheetCompleted.getRange(1, 1, 1, sheetCompleted.getLastColumn()).getValues()[0];
      var cKeyIdx = -1;
      for (var ch = 0; ch < cHeaders.length; ch++) {
        var chStr = String(cHeaders[ch]).trim().toLowerCase();
        if (chStr.includes("sc order no") || chStr.includes("csrm") || chStr.includes("order id tsel")) {
          cKeyIdx = ch + 1;
          break;
        }
      }
      if (cKeyIdx === -1) cKeyIdx = 4;
      var completedKeys = sheetCompleted.getRange(2, cKeyIdx, sheetCompleted.getLastRow() - 1, 1).getValues();
      for (var ci = 0; ci < completedKeys.length; ci++) {
        var cValC = String(completedKeys[ci][0]).trim();
        if (cValC && !KEY_NAMES.some(kn => kn.toLowerCase() === cValC.toLowerCase())) {
          existingKeys[cValC] = true;
        }
      }
    }

    var sourceLastRow = sourceSheet.getLastRow();
    if (sourceLastRow < 2) {
      return { success: true, added: 0, message: "Sheet 'Update Data' kosong." };
    }
    
    var sourceHeaders = sourceSheet.getRange(1, 1, 1, sourceSheet.getLastColumn()).getValues()[0];
    var sourceData = sourceSheet.getRange(2, 1, sourceLastRow - 1, sourceSheet.getLastColumn()).getValues();
    
    var sourceKeyIndex = -1;
    for (var sh = 0; sh < sourceHeaders.length; sh++) {
      var shStr = String(sourceHeaders[sh]).trim().toLowerCase();
      if (shStr.includes("sc order no") || shStr.includes("csrm") || shStr.includes("order id tsel")) {
        sourceKeyIndex = sh;
        break;
      }
    }
    if (sourceKeyIndex === -1) sourceKeyIndex = destKeyIndex - 1;

    var headerMap = {};
    destHeaders.forEach(function(header) {
      var hClean = String(header).trim().toLowerCase();
      if (hClean !== "") {
        var foundIdx = -1;
        for (var s = 0; s < sourceHeaders.length; s++) {
          var sClean = String(sourceHeaders[s]).trim().toLowerCase();
          if (sClean === hClean || 
             ((hClean.includes("order id tsel") || hClean.includes("sc order no")) && 
              (sClean.includes("order id tsel") || sClean.includes("sc order no") || sClean.includes("csrm")))) {
            foundIdx = s;
            break;
          }
        }
        headerMap[header] = foundIdx;
      }
    });

    var rowsToAppend = [];
    
    for (var j = 0; j < sourceData.length; j++) {
      var sourceRow = sourceData[j];
      var keyValue = sourceRow[sourceKeyIndex];
      
      if (keyValue === undefined || keyValue === null) continue;
      var keyString = String(keyValue).trim();
      
      if (keyString === "" || KEY_NAMES.some(kn => kn.toLowerCase() === keyString.toLowerCase())) {
        continue;
      }
      
      if (!existingKeys[keyString]) {
        var newRow = [];
        destHeaders.forEach(function(header) {
          if (String(header).trim() !== "") {
            var srcIdx = headerMap[header];
            if (srcIdx !== undefined && srcIdx !== -1 && srcIdx < sourceRow.length) {
              newRow.push(sourceRow[srcIdx]);
            } else {
              newRow.push("");
            }
          } else {
            newRow.push("");
          }
        });
        rowsToAppend.push(newRow);
        existingKeys[keyString] = true;
      }
    }

    if (rowsToAppend.length > 0) {
      destSheet.getRange(destLastRow + 1, 1, rowsToAppend.length, destHeaders.length).setValues(rowsToAppend);
      return { success: true, added: rowsToAppend.length, message: `${rowsToAppend.length} order baru berhasil ditambahkan ke List Daily Order!` };
    } else {
      return { success: true, added: 0, message: "Semua order sudah sinkron." };
    }
  } catch(e) {
    return { success: false, message: "Gagal sinkronisasi: " + e.message };
  }
}

function getRecentImports() {

}

      existingHeaders = newHeaders;
    }

    let trackIdIdxOld = existingHeaders.findIndex(h => 
      h.includes('sc order no') || h.includes('csrm') || h.includes('order id tsel')
    );

    let rowMap = new Map();
    if (trackIdIdxOld !== -1) {
      existingRows.forEach(row => {
        let tVal = String(row[trackIdIdxOld] || "").trim();
        if (tVal) rowMap.set(tVal, row);
      });
    }

    let newRows = rowsArray.slice(1);
    let addedCount = 0;
    let replacedCount = 0;

    newRows.forEach(newRow => {
      let tNew = trackIdIdxNew !== -1 ? String(newRow[trackIdIdxNew] || "").trim() : "";
      if (tNew && rowMap.has(tNew)) {
        rowMap.set(tNew, newRow);
        replacedCount++;
      } else if (tNew) {
        rowMap.set(tNew, newRow);
        addedCount++;
      } else {
        existingRows.push(newRow);
        addedCount++;
      }
    });

    let finalRows = Array.from(rowMap.values());
    
    sheet.clearContents();
    let finalDataToSet = [rowsArray[0], ...finalRows];
    sheet.getRange(1, 1, finalDataToSet.length, finalDataToSet[0].length).setValues(finalDataToSet);

    let historySheet = ss.getSheetByName("Import History");
    if (!historySheet) {
      historySheet = ss.insertSheet("Import History");
      historySheet.appendRow(["Timestamp", "File Name", "Type", "Status", "Uploader"]);
    }
    let nowStr = new Date().toLocaleString('id-ID');
    historySheet.appendRow([nowStr, fileName, dataType, "Success", uploader]);

    return { 
      success: true, 
      message: `Data berhasil di-import ke sheet 'Update Data'! (${addedCount} baru, ${replacedCount} ter-update karena duplikat).` 
    };
  } catch(e) {
    return { success: false, message: "Gagal import: " + e.message };
  }
}
