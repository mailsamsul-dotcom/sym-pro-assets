        function applyRoleAccess(role) {

            const manageUserMenu = document.getElementById('menu-manage-user');
            if (manageUserMenu) {
                if (roleLower === 'superadmin') {

                }
            }

        window.onload = function() {
            const savedUser = localStorage.getItem("sym_pro_user");
            const savedRole = localStorage.getItem("sym_pro_role");

            if (savedUser) {
                document.getElementById('auth-bg').style.display = 'none';
                document.getElementById('auth-card-container').style.display = 'none';
                document.getElementById('dashboard-wrapper').classList.remove('hidden');
                document.getElementById('user-role-badge').innerText = "Role: " + (savedRole || "USER").toUpperCase();
                
                applyRoleAccess(savedRole || "user");
                switchMenu('daily-order'); 
                loadRecentImports();

            fetchMasterData();
            loadMappingTimData();

        function fetchGlobalUnassignedBadge() {
            if (typeof google === 'undefined' || !google.script || !google.script.run) return;
            google.script.run.withSuccessHandler(function(res) {
                if(res && res.success && res.rows) {
                    updateUnassignedBadgeExplicit(res.headers, res.rows);
                }
            }).getDailyOrderData();
        }

        function updateUnassignedBadgeExplicit(headers, rows) {
            if(!headers || !rows) return;
            let leaderIdx = headers.findIndex(h => String(h).trim().toUpperCase() === 'LEADER');
            let memberIdx = headers.findIndex(h => String(h).trim().toUpperCase() === 'MEMBER');
            
            let unassignedCount = 0;
            rows.forEach(r => {
                let l = leaderIdx !== -1 ? String(r[leaderIdx]).trim() : "-";
                let m = memberIdx !== -1 ? String(r[memberIdx]).trim() : "-";
                if (!l || l === "-" || !m || m === "-") {
                    unassignedCount++;
                }
            });

            const badge = document.getElementById('unassigned-badge');
            if (badge) {
                if (unassignedCount > 0) {
                    badge.innerText = unassignedCount;
                    badge.classList.remove('hidden');
                } else {
                    badge.classList.add('hidden');
                }
            }
        }

        function fetchMasterData() {
            if (typeof google === 'undefined' || !google.script || !google.script.run) return;

            google.script.run.withSuccessHandler(function(res) {
                if(res.success) {
                    techniciansCache = res.technicians;
                    populateTechSelectOptions();
                }
            }).getMasterTechnicians();

            google.script.run.withSuccessHandler(function(res) {
                if(res.success) errorCodesCache = res.errorCodes;
            }).getMasterErrorCodes();
        }

        function populateTechSelectOptions() {
            const mLeader = document.getElementById('map-leader');
            const mMember = document.getElementById('map-member');
            if(!mLeader || !mMember) return;
            
            let assignedTechs = new Set();
            mappingTimCache.forEach(m => {
                if(m.leader) assignedTechs.add(m.leader);
                if(m.member) assignedTechs.add(m.member);
            });

            let availableTechs = techniciansCache.filter(t => !assignedTechs.has(t));
            let opts = `<option value="">-- Pilih --</option>`;
            availableTechs.forEach(t => { opts += `<option value="${t}">${t}</option>`; });
            
            mLeader.innerHTML = opts;
            mMember.innerHTML = opts;
        }

        function loadMappingTimData() {
            const tbody = document.getElementById('mapping-tim-body');
            if(!tbody) return;
            
            if (typeof google === 'undefined' || !google.script || !google.script.run) return;
            tbody.innerHTML = `<tr><td colspan="4" class="text-center py-6 text-gray-400"><i class="fa-solid fa-spinner fa-spin mr-2"></i> Memuat tim...</td></tr>`;

            google.script.run.withSuccessHandler(function(res) {
                if(res.success) {
                    mappingTimCache = res.data;
                    renderMappingTimTable();
                    populateTechSelectOptions();
                } else {
                    tbody.innerHTML = `<tr><td colspan="4" class="text-center py-6 text-red-500">${res.message}</td></tr>`;
                }
            }).getMappingTim();
        }

        function renderMappingTimTable() {
            const tbody = document.getElementById('mapping-tim-body');
            if(mappingTimCache.length === 0) {
                tbody.innerHTML = `<tr><td colspan="4" class="text-center py-6 text-gray-400">Belum ada mapping tim.</td></tr>`;
                return;
            }
            let html = "";
            mappingTimCache.forEach(item => {
                html += `
                    <tr class="hover:bg-gray-50 transition border-b border-gray-50">
                        <td class="py-2.5 px-4 font-bold text-gray-800">${item.tim}</td>
                        <td class="py-2.5 px-4 text-gray-600">${item.leader}</td>
                        <td class="py-2.5 px-4 text-gray-600">${item.member}</td>
                        <td class="py-2.5 px-4 text-center">
                            <button onclick="deleteMappingTim('${item.tim}')" class="px-2.5 py-1 bg-red-50 hover:bg-red-100 text-red-600 rounded-lg text-[10px] font-bold transition"><i class="fa-solid fa-trash"></i></button>
                        </td>
                    </tr>
                `;
            });
            tbody.innerHTML = html;
        }

            const tim = document.getElementById('map-tim-name').value;
            const leader = document.getElementById('map-leader').value;
            const member = document.getElementById('map-member').value;

            if(!tim || !leader || !member) {
                showMessage("Lengkapi Nama Tim, Leader, dan Member!", "error"); return;

                        document.getElementById('map-tim-name').value = "";

                        loadMappingTimData();
                    } else {

                }).saveMappingTim(tim, leader, member);

        }

        function showMessage(msg, type = 'error') {

            toast.style.opacity = '1';

        }

        function togglePassword(inputId, iconElement) {

        function populateMultiSelectFilters() {
            if(!globalDailyHeaders || globalDailyHeaders.length === 0) return;
            let crmTypeIdx = -1, jenisErrorIdx = -1, errorcodeIdx = -1;
            globalDailyHeaders.forEach((h, idx) => {
                let hClean = String(h).trim().toLowerCase();
                if(hClean === 'crm order type') crmTypeIdx = idx;
                if(hClean === 'jenis_error') jenisErrorIdx = idx;
                if(hClean === 'errorcode') errorcodeIdx = idx;
            });

            const crmContent = document.getElementById('content-crm'); crmContent.innerHTML = "";
            if(crmTypeIdx !== -1) {
                let uniqueCrm = [...new Set(globalDailyRows.map(r => String(r[crmTypeIdx] || "").trim()))].filter(v => v !== "");
                uniqueCrm.sort().forEach(val => { crmContent.innerHTML += `<label class="multiselect-item"><input type="checkbox" value="${val}" class="custom-checkbox cb-crm" onchange="updateDropdownLabel('crm')"><span>${val}</span></label>`; });
            } updateDropdownLabel('crm');

            const jenisContent = document.getElementById('content-jenis'); jenisContent.innerHTML = "";
            if(jenisErrorIdx !== -1) {
                let uniqueJenis = [...new Set(globalDailyRows.map(r => String(r[jenisErrorIdx] || "").trim()))].filter(v => v !== "");
                uniqueJenis.sort().forEach(val => { jenisContent.innerHTML += `<label class="multiselect-item"><input type="checkbox" value="${val}" class="custom-checkbox cb-jenis" onchange="updateDropdownLabel('jenis')"><span>${val}</span></label>`; });
            } updateDropdownLabel('jenis');

            const errContent = document.getElementById('content-errorcode'); errContent.innerHTML = "";
            if(errorcodeIdx !== -1) {
                let uniqueErr = [...new Set(globalDailyRows.map(r => String(r[errorcodeIdx] || "").trim()))].filter(v => v !== "");
                uniqueErr.sort().forEach(val => { errContent.innerHTML += `<label class="multiselect-item"><input type="checkbox" value="${val}" class="custom-checkbox cb-errorcode" onchange="updateDropdownLabel('errorcode')"><span>${val}</span></label>`; });
            } updateDropdownLabel('errorcode');
        }

        function applyFilters() {
            if(!globalDailyHeaders || globalDailyHeaders.length === 0) {
                renderTable([]);
                return;
            }

            const searchQuery = document.getElementById('filter-search').value.toLowerCase().trim();
            const selectedCrms = Array.from(document.querySelectorAll('.cb-crm:checked')).map(cb => cb.value);
            const selectedJenis = Array.from(document.querySelectorAll('.cb-jenis:checked')).map(cb => cb.value);
            const selectedErrors = Array.from(document.querySelectorAll('.cb-errorcode:checked')).map(cb => cb.value);
            const dateFrom = document.getElementById('filter-date-from').value;
            const dateTo = document.getElementById('filter-date-to').value;

            let workorderIdx = -1, trackIdIdx = -1, crmTypeIdx = -1, jenisErrorIdx = -1, errorcodeIdx = -1, bookingDateIdx = -1;

            globalDailyHeaders.forEach((h, idx) => {
                let hClean = String(h).trim().toLowerCase();
                if(hClean.includes('workorder')) workorderIdx = idx;
                if(hClean.includes('sc order no') || hClean.includes('csrm')) trackIdIdx = idx;
                if(hClean === 'crm order type') crmTypeIdx = idx;
                if(hClean === 'jenis_error') jenisErrorIdx = idx;
                if(hClean === 'errorcode') errorcodeIdx = idx;
                if(hClean === 'booking date') bookingDateIdx = idx;
            });

            let filtered = globalDailyRows.filter(row => {
                let matchSearch = true;
                if(searchQuery !== "") {
                    let woVal = workorderIdx !== -1 ? String(row[workorderIdx] || "").toLowerCase() : "";
                    let trVal = trackIdIdx !== -1 ? String(row[trackIdIdx] || "").toLowerCase() : "";
                    matchSearch = woVal.includes(searchQuery) || trVal.includes(searchQuery);
                }

                let matchCrm = true;
                if(selectedCrms.length > 0 && crmTypeIdx !== -1) matchCrm = selectedCrms.includes(String(row[crmTypeIdx] || "").trim());

                let matchJenis = true;
                if(selectedJenis.length > 0 && jenisErrorIdx !== -1) matchJenis = selectedJenis.includes(String(row[jenisErrorIdx] || "").trim());

                let matchErrorcode = true;
                if(selectedErrors.length > 0 && errorcodeIdx !== -1) matchErrorcode = selectedErrors.includes(String(row[errorcodeIdx] || "").trim());

                let matchDate = true;
                if((dateFrom || dateTo) && bookingDateIdx !== -1) {
                    let cellDateVal = row[bookingDateIdx];
                    if(cellDateVal) {
                        let rowDate = new Date(cellDateVal.split(',')[0]);
                        if(!isNaN(rowDate.getTime())) {
                            rowDate.setHours(0,0,0,0);
                            if(dateFrom) {
                                let fDate = new Date(dateFrom); fDate.setHours(0,0,0,0);
                                if(rowDate < fDate) matchDate = false;
                            }
                            if(dateTo) {
                                let tDate = new Date(dateTo); tDate.setHours(0,0,0,0);
                                if(rowDate > tDate) matchDate = false;
                            }
                        }
                    }
                }
                return matchSearch && matchCrm && matchJenis && matchErrorcode && matchDate;
            });

            renderTable(filtered);
        }

        function resetFilters() {
            document.getElementById('filter-search').value = "";
            document.querySelectorAll('.multiselect-dropdown input[type="checkbox"]').forEach(cb => cb.checked = false);
            updateDropdownLabel('crm'); updateDropdownLabel('jenis'); updateDropdownLabel('errorcode');
            document.getElementById('filter-date-from').value = ""; document.getElementById('filter-date-to').value = "";
            applyFilters(); showMessage("Filter berhasil direset.", "success");
        }

        function renderTable(rowsToRender) {
            const thead = document.getElementById('daily-order-head');
            const tbody = document.getElementById('daily-order-body');
            
            function getColVal(row, colName) {
                let idx = globalDailyHeaders.findIndex(h => String(h).trim().toLowerCase() === colName.toLowerCase());
                return (idx !== -1 && row[idx] !== undefined && row[idx] !== null) ? row[idx] : "";
            }

            let headHtml = `<tr class="text-[11px] uppercase tracking-wider">`;
            headHtml += `<th class="py-3 px-3 text-center sticky-head-aksi">Aksi</th>`;
            TARGET_COLUMNS.forEach(colName => {
                if (colName === "Workorder") headHtml += `<th class="py-3 px-3 whitespace-nowrap sticky-head-wo">${colName}</th>`;
                else headHtml += `<th class="py-3 px-3 whitespace-nowrap sticky-head-other">${colName}</th>`;
            }); 
            headHtml += `</tr>`;
            thead.innerHTML = headHtml;

            let validRows = rowsToRender.filter(row => {
                let wo = String(getColVal(row, "Workorder")).trim();
                let track = String(getColVal(row, "SC Order No/Track ID/CSRM No")).trim();
                let cust = String(getColVal(row, "Customer Name")).trim();

                let custValid = cust !== "" && !cust.toUpperCase().includes("CUSTOMER NAME");
                
                return woValid || trackValid || custValid;
            });

            document.getElementById('table-row-count').innerText = `Menampilkan ${validRows.length} dari ${globalDailyRows.length} data`;

            if(validRows.length === 0) {
                tbody.innerHTML = `<tr><td colspan="${TARGET_COLUMNS.length + 1}" class="text-center py-6 text-gray-400">Tidak ada data yang cocok atau data sheet kosong.</td></tr>`;
                return;
            }

            let bodyHtml = "";

                bodyHtml += `<td class="py-2.5 px-3 text-center sticky-aksi"><button class="px-2.5 py-1 bg-blue-100 hover:bg-blue-200 text-[#0d55d4] rounded-lg font-medium text-[11px] transition"><i class="fa-solid fa-eye mr-1"></i> Detail</button></td>`;

                        let errPart = (err || subErr) ? `${err} - ${subErr}` : ""; let errHtml = errPart ? `<span class="text-red-600 font-bold">${errPart}</span>` : "";
                        let memoHtml = memo ? `<span class="italic text-gray-600">${memo}</span>` : "";

            });

        function saveOrderDetail(actionType, timData = null) {
            const rowData = globalDailyRows[currentEditingRowIndex];
            
            function getVal(colName) {
                let idx = globalDailyHeaders.findIndex(h => String(h).trim().toLowerCase() === colName.toLowerCase());
                return (idx !== -1 && rowData[idx] !== undefined && rowData[idx] !== null) ? String(rowData[idx]).trim() : "";
            }

            let sheetTargetName = 'List Daily Order';
            if (currentActiveMode === 'order-kendala') sheetTargetName = 'Order Kendala';
            else if (currentActiveMode === 'order-completed') sheetTargetName = 'Order Completed';

            let oldWo = getVal("Workorder");
            let oldTr = getVal("SC Order No/Track ID/CSRM No");
            if (oldTr === "-") oldTr = "";

            let finalLeader = getVal("LEADER");
            let finalMember = getVal("MEMBER");
            let finalTimName = "";
            let finalTimestamp = null;
            
            if (actionType === 'assign' && timData) {
                finalLeader = timData.leader;
                finalMember = timData.member;
                finalTimName = timData.tim;

                const now = new Date();
                const dd = String(now.getDate()).padStart(2, '0');
                const mm = String(now.getMonth() + 1).padStart(2, '0');
                const yyyy = now.getFullYear();
                const hh = String(now.getHours()).padStart(2, '0');
                const min = String(now.getMinutes()).padStart(2, '0');
                finalTimestamp = `${dd}/${mm}/${yyyy} ${hh}:${min}`;

                let assignStampHtml = `<div class="mt-1.5 text-[10px] text-gray-400 italic"><i class="fa-regular fa-clock"></i> Diproses: ${finalTimestamp}</div>`;
                document.getElementById('display-techs').innerHTML = `<i class="fa-solid fa-user-tie text-blue-600 mr-1"></i> ${finalLeader} <span class="text-gray-400 mx-1">/</span> <i class="fa-solid fa-users text-gray-600 mr-1"></i> ${finalMember}`;
                document.getElementById('assign-stamp-container').innerHTML = assignStampHtml;
                
                document.getElementById('btn-assign-container').innerHTML = `<button onclick="openTechnicianSelector()" class="px-5 py-2 bg-amber-500 hover:bg-amber-600 text-white font-semibold rounded-xl text-xs transition shadow-xs flex items-center gap-1.5 cursor-pointer"><i class="fa-solid fa-rotate"></i> ReAssign</button>`;
            }

            let payload = {
                sheetName: sheetTargetName,
                action: actionType,
                oldWorkorder: oldWo,
                oldTrackId: oldTr,
                workorder: isOrderEditMode && document.getElementById('edit-wo') ? document.getElementById('edit-wo').value : oldWo,
                trackId: isOrderEditMode && document.getElementById('edit-track') ? document.getElementById('edit-track').value : oldTr,
                crmOrderType: isOrderEditMode && document.getElementById('edit-crm') ? document.getElementById('edit-crm').value : getVal("CRM Order Type"),
                serviceNo: isOrderEditMode && document.getElementById('edit-service') ? document.getElementById('edit-service').value : getVal("Service No."),
                packageInfo: isOrderEditMode && document.getElementById('edit-pkg') ? document.getElementById('edit-pkg').value : getVal("PACKAGE_INFO"),
                leader: finalLeader,
                member: finalMember,
                timName: finalTimName,
                assignTime: finalTimestamp,
                errorCode: document.getElementById('input-errcode') ? document.getElementById('input-errcode').value : getVal("ERRORCODE"),
                subErrorCode: document.getElementById('input-suberr') ? document.getElementById('input-suberr').value : getVal("SUBERRORCODE"),
                engineerMemo: document.getElementById('input-memo') ? document.getElementById('input-memo').value : getVal("ENGINEERMEMO (SA)")
            };

            if (typeof google === 'undefined' || !google.script || !google.script.run) return;
            google.script.run
            .withSuccessHandler(function(res) {
                showMessage(res.message, res.success ? 'success' : 'error');
                if(res.success) {
                    if(actionType !== 'assign') closeDetailModal();
                    loadActiveOrderData();
                    fetchGlobalUnassignedBadge();
                }
            })
            .updateDailyOrderData(payload);
        }

        function exportExcelActive() {
            if(!globalDailyHeaders || globalDailyHeaders.length === 0) return;
            let dataToExport = [globalDailyHeaders, ...globalDailyRows];
            let ws = XLSX.utils.aoa_to_sheet(dataToExport); let wb = XLSX.utils.book_new();
            let exportSheetName = currentActiveMode === 'order-kendala' ? "Order Kendala" : (currentActiveMode === 'order-completed' ? "Order Completed" : "List Daily Order");
            XLSX.utils.book_append_sheet(wb, ws, exportSheetName);
            XLSX.writeFile(wb, `${exportSheetName}_Export.xlsx`);
        }

        function loadRecentImports() {
            if (typeof google === 'undefined' || !google.script || !google.script.run) return;
            google.script.run.withSuccessHandler(function(historyList) {
                const tbody = document.getElementById('history-table-body');
                if (!historyList || historyList.length === 0) {
                    tbody.innerHTML = `<tr><td class="py-3 px-3 text-gray-400 italic" colspan="6">Belum ada riwayat impor.</td></tr>`;
                    return;
                }
                tbody.innerHTML = "";

                    tr.className = "hover:bg-gray-50 transition border-b border-gray-50";
                    tr.innerHTML = `
                        <td class="py-3 px-3 font-semibold text-gray-700">${index + 1}</td>
                        <td class="py-3 px-3 font-medium text-gray-900">${item.file}</td>
                        <td class="py-3 px-3"><span class="bg-blue-50 text-blue-700 px-2 py-0.5 rounded font-bold">${item.type}</span></td>
                        <td class="py-3 px-3"><span class="bg-emerald-100 text-emerald-700 text-[10px] font-bold px-2 py-0.5 rounded">${item.status}</span></td>
                        <td class="py-3 px-3 text-gray-600">${item.by}</td>
                        <td class="py-3 px-3 text-gray-400">${item.timestamp}</td>
                    `;
                    tbody.appendChild(tr);
                });
            }).getRecentImports();
        }

        function processUploadExcel() {
            const fileInput = document.getElementById('excel-file'); 
            const dataType = document.getElementById('import-type').value;
            if (fileInput.files.length === 0) { 
                showMessage("Pilih file Excel terlebih dahulu!", "error"); 
                return; 
            }
            const file = fileInput.files[0]; 
            const uploader = localStorage.getItem("sym_pro_user") || "User";
            const reader = new FileReader(); 
            document.getElementById('import-loading').classList.remove('hidden');

            reader.onload = function(e) {
                try {
                    const data = new Uint8Array(e.target.result); 
                    const workbook = XLSX.read(data, { type: 'array' });
                    const worksheet = workbook.Sheets[workbook.SheetNames[0]]; 
                    const rowsArray = XLSX.utils.sheet_to_json(worksheet, { header: 1, defval: "" });
                    if (rowsArray.length <= 1) { 
                        document.getElementById('import-loading').classList.add('hidden'); 
                        showMessage("File kosong.", "error"); 
                        return; 
                    }
                    if (typeof google === 'undefined' || !google.script || !google.script.run) return;

                    google.script.run.withSuccessHandler(function(res) {
                        document.getElementById('import-loading').classList.add('hidden');
                        if (res.success) {
                            showMessage(res.message, 'success'); 
                            fileInput.value = ''; 
                            document.getElementById('file-chosen').innerText = "No file chosen"; 
                            loadRecentImports();
                            fetchGlobalUnassignedBadge();
                        } else {
                            showMessage(res.message, 'error');
                        }
                    }).importExcelData(rowsArray, dataType, uploader, file.name);
                } catch (err) {
                    document.getElementById('import-loading').classList.add('hidden'); 
                    showMessage("Gagal membaca Excel", "error");
                }
            }; 
            reader.readAsArrayBuffer(file);
        }
